# 13. 数组去重

一、利用ES6的Set数据结构的特性

Set集合里的所有的元素都是唯一的

var arr = [1,1,'true','true',true,true,15,15,false,false,undefined,undefined,null,null,NaN,NaN,'NaN',0, 0,'a','a',{},{}];
function unique (arr) {
    // Array.from()方法 (es6) 就是将一个类数组对象或者可遍历对象转换成一个真正的数组
    // 或 [...new Set(arr)] 
    return Array.from(new Set(arr))
}
console.log(unique(arr))
// [1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {}, {}]

缺点：

不能过滤重复的 Object

二、利用ES6的Map数据结构的特性

var arr = [1,1,'true','true',true,true,15,15,false,false,undefined,undefined,null,null,NaN,NaN,'NaN',0, 0,'a','a',{},{}];
function unique(arr) {
    let map = new Map()
    let newArr = []
    for(let i = 0; i < arr.length; i++) {
        if(!map.has(arr[i])) {
            map.set(arr[i])
            newArr.push(arr[i])
        }
    }
    return newArr
}
console.log(unique(arr))
// [1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {}, {}]

缺点：

不能过滤重复的 Object

三、利用对象的属性key唯一的特性去重

var arr = [1,1,'true','true',true,true,15,15,false,false,undefined,undefined,null,null,NaN,NaN,'NaN',0, 0,'a','a',{},{}];
function unique(arr) {
    let obj = {}
    let newArr = []
    for(let i = 0; i < arr.length; i++) {
        if(!obj[arr[i]]){
            obj[arr[i]] = 1
            newArr.push(arr[i])
        }
    }
    return newArr
}
console.log(unique(arr))
// [1, "true", 15, false, undefined, null, NaN, 0, "a", {}]

缺点：

针对 NaN和'NaN', 对象的key会视为一个key，区分不了NaN和'NaN'。

四、indexOf去重

function distinct (arr) {
    let newArr = []
    for(let i = 0; i < arr.length; i++) {
        if(newArr.indexOf(arr[i]) < 0) {
            newArr.push(arr[i])
        } 
    }
    return newArr
}
// [1, "true", true, 15, false, undefined, null, NaN, NaN, "NaN", 0, "a", {…}, {…}]

缺点：

不能过滤掉 NaN、Object

五、includes去重

function distinct (arr) {
    let newArr = []
    for(let i = 0; i < arr.length; i++) {
        if(!newArr.includes(arr[i])) {
            newArr.push(arr[i])
        } 
    }
    return newArr
}
// [1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {}, {}]

缺点：

不能过滤掉 Object

六、利用hasOwnProperty

// 利用hasOwnProperty 判断是否存在对象属性
function unique(arr) {
    var obj = {};
    return arr.filter(function(item, index, arr){
        return obj.hasOwnProperty(typeof item + item) ? false : (obj[typeof item + item] = true)
    })
}
// 所有的都去重了
// [1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {}]

七、利用filter

function unique(arr) {
    return arr.filter(function(item, index, arr) {
        //当前元素，在原始数组中的第一个索引==当前索引值，否则返回当前元素
        return arr.indexOf(item, 0) === index;
    });
}
// [1, "true", true, 15, false, undefined, null, "NaN", 0, "a", {}, {}]

缺点：

不能过滤掉 Object

八、reduce数组去重

function distinct(arr) {
    return arr.sort().reduce((init, current) => {
        if(init.length === 0 || init[init.length-1] !== current) {
            init.push(current);
        }
        return init;
    }, []);
}
// [0, 1, 15, "NaN", "a", false, null, "true", true, undefined, NaN, NaN, {}, {}]
  • 该方法先对数组进行排序，在进行去重过滤，为了将一样的元素放在一起

缺点：

不能过滤掉 NaN、Object
